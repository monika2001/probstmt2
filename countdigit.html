Program for Counts the digits in a number

#include using namespace std;
// Function to counts the digits in a number
int countDigit(long long n)
{
int count = 0;
while (n != 0)
{
n = n / 10;
++count;
}
return count;
}
int main(void)
{
long long n = 345289467;
cout << "Number of digits : " << countDigit(n);
return 0;
}
